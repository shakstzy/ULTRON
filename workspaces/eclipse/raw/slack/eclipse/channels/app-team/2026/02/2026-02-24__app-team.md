> ### 13:51 — daniel
> 
> ```name: Prisma Baseline Manager
>
> on:
>   workflow_dispatch:
>     inputs:
>       action:
>         description: "What to run against the selected database"
>         required: true
>         type: choice
>         options:
>           - resolve
>           - deploy
>       environment:
>         description: "GitHub environment that provides DATABASE_URL"
>         required: true
>         type: choice
>         options:
>           - development
>           - production
>       migration_name:
>         description: "Migration folder name to resolve as applied (used only for resolve)"
>         required: false
>         default: "0_init"
>         type: string
>
> jobs:
>   prisma-baseline:
>     name: Prisma ${{ inputs.action }} (${{ inputs.environment }})
>     runs-on: ubuntu-latest
>     environment: ${{ inputs.environment }}
>     permissions:
>       contents: read
>
>     steps:
>       - name: Checkout
>         uses: actions/checkout@v4
>
>       - name: Setup pnpm
>         uses: pnpm/action-setup@v4
>
>       - name: Setup Node
>         uses: actions/setup-node@v4
>         with:
>           node-version: 22
>           cache: pnpm
>
>       - name: Install dependencies
>         run: pnpm install --frozen-lockfile
>
>       - name: Validate baseline migration exists
>         if: ${{ inputs.action == 'resolve' }}
>         run: |
>           if [ ! -d "prisma/migrations/${{ inputs.migration_name }}" ]; then
>             echo "::error::Migration folder prisma/migrations/${{ inputs.migration_name }} not found."
>             exit 1
>           fi
>
>       - name: Resolve baseline as applied (one-time per existing DB)
>         if: ${{ inputs.action == 'resolve' }}
>         env:
>           DATABASE_URL: ${{ secrets.DATABASE_URL }}
>         run: pnpm prisma migrate resolve --applied "${{ inputs.migration_name }}"
>
>       - name: Deploy migrations
>         if: ${{ inputs.action == 'deploy' }}
>         env:
>           DATABASE_URL: ${{ secrets.DATABASE_URL }}
>         run: pnpm prisma migrate deploy```
